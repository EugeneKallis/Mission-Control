from fastapi import FastAPI, HTTPException, Request
from fastapi.responses import HTMLResponse, JSONResponse
from fastapi.staticfiles import StaticFiles
from fastapi.templating import Jinja2Templates
import uvicorn
import subprocess
import json
from datetime import datetime
from typing import Optional

app = FastAPI(title="Hermes Mission Control")
app.mount("/static", StaticFiles(directory="static"), name="static")
templates = Jinja2Templates(directory="templates")


def run_cronjob_action(action: str, job_id: Optional[str] = None, **kwargs):
    """Execute cronjob commands via subprocess."""
    cmd = ["python", "-c"]
    
    script = f"""
import sys
sys.path.insert(0, '/home/ponzi/.hermes')
from cronjobs.manager import CronJobManager
manager = CronJobManager()
"""
    
    if action == "list":
        script += "result = manager.list_jobs(include_disabled=True)\nprint(json.dumps(result, default=str))"
    elif action == "get" and job_id:
        script += f"result = manager.get_job('{job_id}')\nprint(json.dumps(result, default=str))" if job_id else "pass"
    elif action == "pause" and job_id:
        reason = kwargs.get('reason', 'Paused via UI')
        script += f"manager.pause_job('{job_id}', reason='{reason}')\nprint('{{\"success\": true}}')"
    elif action == "resume" and job_id:
        script += f"manager.resume_job('{job_id}')\nprint('{{\"success\": true}}')"
    elif action == "update" and job_id:
        # Build update dict
        updates = []
        for key, val in kwargs.items():
            updates.append(f"'{key}': {repr(val)}")
        update_str = ", ".join(updates)
        script += f"manager.update_job('{job_id}', {{{update_str}}})\nprint('{{\"success\": true}}')"
    else:
        return {"error": f"Unknown action: {action}"}
    
    cmd.append(script)
    
    try:
        result = subprocess.run(cmd, capture_output=True, text=True, timeout=30, cwd="/home/ponzi/.hermes")
        if result.returncode != 0:
            return {"error": result.stderr}
        return json.loads(result.stdout.strip().split('\n')[-1])
    except Exception as e:
        return {"error": str(e)}


@app.get("/", response_class=HTMLResponse)
async def index(request: Request):
    return templates.TemplateResponse("index.html", {"request": request})


@app.get("/crons", response_class=HTMLResponse)
async def crons_page(request: Request):
    return templates.TemplateResponse("crons.html", {"request": request})


@app.get("/api/crons")
async def list_crons():
    """List all cron jobs."""
    jobs = run_cronjob_action("list")
    return jobs


@app.get("/api/crons/{job_id}")
async def get_cron(job_id: str):
    """Get a specific cron job."""
    result = run_cronjob_action("list")
    if "jobs" in result:
        for job in result["jobs"]:
            if job.get("job_id") == job_id:
                return job
    raise HTTPException(status_code=404, detail="Job not found")


@app.post("/api/crons/{job_id}/pause")
async def pause_cron(job_id: str):
    """Pause a cron job."""
    result = run_cronjob_action("pause", job_id=job_id)
    return result


@app.post("/api/crons/{job_id}/resume")
async def resume_cron(job_id: str):
    """Resume a cron job."""
    result = run_cronjob_action("resume", job_id=job_id)
    return result


@app.patch("/api/crons/{job_id}")
async def update_cron(job_id: str, request: Request):
    """Update a cron job."""
    data = await request.json()
    result = run_cronjob_action("update", job_id=job_id, **data)
    return result


if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8765)
